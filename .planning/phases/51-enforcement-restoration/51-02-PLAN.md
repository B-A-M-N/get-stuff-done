```javascript
function persistItlOutput(phaseNumber, phaseSlug, itlResult) {
  const dir = path.join(process.cwd(), '.planning', 'phases', `${String(phaseNumber).padStart(2,'0')}-${phaseSlug}`);
  fs.mkdirSync(dir, { recursive: true });
  const file = path.join(dir, `${String(phaseNumber).padStart(2,'0')}-ITL.json`);

    lockability: itlResult.lockability,
    interpretation: itlResult.interpretation,
    summary: itlResult.summary,
    generated_at: new Date().toISOString(),
  };

}
```

<action>

   - Or within `cmdVerifyCheckpointResponse` / execute-phase workflow
