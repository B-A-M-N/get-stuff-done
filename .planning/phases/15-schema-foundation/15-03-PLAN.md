function cmdVerifyCheckpointResponse(cwd, filePath, raw) {
  if (!filePath) { error('file path required'); }
  const fullPath = path.isAbsolute(filePath) ? filePath : path.join(cwd, filePath);
  const content = safeReadFile(fullPath);
  if (!content) { output({ error: 'File not found', path: filePath }, raw); return; }

  const requiredFields = ['status','why_blocked','what_is_uncertain','choices','allow_freeform','resume_condition'];
  const errors = [];
  const extracted = {};

  for (const field of requiredFields) {
    const pattern = new RegExp('^' + field + ':\\s*(.+)$', 'im');
    const match = content.match(pattern);
    if (!match) { errors.push('Missing checkpoint field: ' + field); continue; }
    extracted[field] = match[1].trim();
  }

  // ... validation checks ...

  output({ valid: errors.length === 0, errors, fields: extracted }, raw, errors.length === 0 ? 'valid' : 'invalid');
}
