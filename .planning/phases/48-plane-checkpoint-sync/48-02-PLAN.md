    Step A: Create `get-stuff-done/bin/lib/summary-plane-sync.cjs` with:
    ```javascript
    const path = require('path');
    const { safeFs, logWarn, findPhaseInternal } = require('./core.cjs');
    const planeClient = require('./plane-client.cjs');
    const { findIssueByCustomField } = require('./roadmap-plane-sync.cjs');

    async function notifySummaryWrite(cwd, phase, plan) {
      if (!planeClient.apiKey || !planeClient.projectId) {
        return;
      }
      try {
        const phaseInfo = findPhaseInternal(cwd, phase);
        if (!phaseInfo) {
          throw new Error(`Phase ${phase} not found`);
        }
        const paddedPhase = String(phase).padStart(2, '0');
        const paddedPlan = String(plan).padStart(2, '0');
        const summaryFileName = `${paddedPhase}-${paddedPlan}-SUMMARY.md`;
        const summaryPath = path.join(cwd, phaseInfo.directory, summaryFileName);
        if (!safeFs.existsSync(summaryPath)) {
          throw new Error(`Summary file not found: ${summaryPath}`);
        }
        const content = safeFs.readFileSync(summaryPath, 'utf-8');

        // Extract frontmatter (between --- lines)
        const fmMatch = content.match(/^---\n([\s\S]*?)\n---/);
        if (!fmMatch) {
          throw new Error('Missing frontmatter in summary');
        }
        const frontmatter = {};
        for (const line of fmMatch[1].split('\n')) {
          const kv = line.split(':', 2);
          if (kv.length === 2) {
            const key = kv[0].trim();
            let val = kv[1].trim();
            try {
              // Parse JSON for arrays/objects if applicable
              val = JSON.parse(val);
            } catch {}
            frontmatter[key] = val;
          }
        }

        // Find the plan issue via gsd_plan_id custom field
        const planId = `${paddedPhase}-${paddedPlan}`;
        const issue = await findIssueByCustomField('gsd_plan_id', planId, { dryRun: false });
        if (!issue) {
          logWarn('Summary comment: no Plane issue found for plan', { plan: planId });
          return;
        }
        const issueId = issue.id || issue.data?.id;
        if (!issueId) {
          logWarn('Summary comment: issue missing ID', { issue });
          return;
        }

        // Format comment with key summary fields
        const comment = `
**Phase Summary**

- **Phase:** ${phase}
- **Plan:** ${planId}
- **Subsystem:** ${frontmatter.subsystem || 'N/A'}
- **Provides:** ${(frontmatter.provides || []).join(', ')}
- **Duration:** ${frontmatter.duration || 'unknown'}
- **Completed:** ${frontmatter.completed || 'unknown'}
- **Requirements:** ${(frontmatter.requirements_completed || []).join(', ')}

*Full details: `.planning/phases/${phaseInfo.directory}/${summaryFileName}`*
`.trim();

        await planeClient.addComment(issueId, comment);
      } catch (err) {
        logWarn('Summary comment failed:', { error: err.message });
      }
    }

    module.exports = { notifySummaryWrite };
    ```

    ```javascript
    case 'plane-sync': {
      const sub = args[1];
      if (sub === 'summary') {
        const cwd = process.cwd();
        const phaseIdx = args.indexOf('--phase');
        const planIdx = args.indexOf('--plan');
        if (phaseIdx === -1 || planIdx === -1) {
          error('plane-sync summary requires --phase and --plan');
        }
        const phase = parseInt(args[phaseIdx + 1], 10);
        const plan = parseInt(args[planIdx + 1], 10);
        if (isNaN(phase) || isNaN(plan)) {
          error('--phase and --plan must be numeric');
        }
        const sync = require('./lib/summary-plane-sync.cjs');
        await sync.notifySummaryWrite(cwd, phase, plan);
        output({ synced: true, phase, plan }, raw, 'synced');
      } else {
        error('Unknown plane-sync subcommand. Available: summary');
      }
      break;
    }
    ```
