    Step A: Create `get-stuff-done/bin/lib/checkpoint-plane-sync.cjs` with the following content:
    ```javascript
    const path = require('path');
    const { safeFs, logWarn } = require('./core.cjs');
    const planeClient = require('./plane-client.cjs');
    const { findIssueByCustomField } = require('./roadmap-plane-sync.cjs');

    async function notifyCheckpointWrite(phase, checkpointPath) {
      if (!planeClient.apiKey || !planeClient.projectId) {
        return; // Plane sync disabled
      }
      try {
        // Read checkpoint file
        const content = safeFs.readFileSync(checkpointPath, 'utf-8');
        const fm = {};
        let body = '';
        const fmMatch = content.match(/^---\n([\s\S]*?)\n---\n([\s\S]*)$/);
        if (fmMatch) {
          const fmText = fmMatch[1];
          body = fmMatch[2].trim();
          for (const line of fmText.split('\n')) {
            const kv = line.split(':', 2);
            if (kv.length === 2) fm[kv[0].trim()] = kv[1].trim();
          }
        } else {
          throw new Error('Invalid checkpoint format');
        }

        // Find the phase issue via custom field gsd_phase_number
        const issue = await findIssueByCustomField('gsd_phase_number', String(phase), { dryRun: false });
        if (!issue) {
          logWarn('Checkpoint comment: no Plane issue found for phase', { phase });
          return;
        }

        const issueId = issue.id || issue.data?.id;
        if (!issueId) {
          logWarn('Checkpoint comment: issue missing ID', { issue });
          return;
        }

        // Format comment with key checkpoint fields
        const comment = `
**Checkpoint artifact**

- **Type:** ${fm.type || 'N/A'}
- **Blocked at:** ${Object.hasOwn(fm, 'taskNum') ? `Task ${fm.taskNum}` : 'N/A'}
- **Why blocked:** ${fm.why_blocked}
- **What is uncertain:** ${fm.what_is_uncertain}
${fm.choices && fm.choices !== 'none' ? `- **Choices:** ${fm.choices}` : ''}
- **Resume condition:** ${fm.resume_condition}

*Artifact: `.planning/phases/${phase}/CHECKPOINT.md`*
`.trim();

        await planeClient.addComment(issueId, comment);
      } catch (err) {
        logWarn('Checkpoint comment failed:', { error: err.message });
      }
    }

    module.exports = { notifyCheckpointWrite };
    ```
